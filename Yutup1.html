    <script>
        function showForm(formId) {
            document.getElementById('signup-form').classList.remove('show');
            document.getElementById('signup-form').classList.add('hide');
            document.getElementById('user-login-form').classList.remove('show');
            document.getElementById('user-login-form').classList.add('hide');
            document.getElementById('admin-login-form').classList.remove('show');
            document.getElementById('admin-login-form').classList.add('hide');
            document.getElementById('comment-form').classList.remove('show');
            document.getElementById('comment-form').classList.add('hide');

            document.getElementById(formId).classList.remove('hide');
            document.getElementById(formId).classList.add('show');
        }

        function signUp() {
            alert('Kayıt olma işlevi henüz uygulanmadı.');
        }

        function login() {
            const username = document.getElementById('login-username').value;
            const password = document.getElementById('login-password').value;

            if (username === 'kullanici' && password === 'sifre') {
                alert('Başarıyla giriş yapıldı.');
                document.getElementById('comment-form').classList.remove('hide');
                document.getElementById('comment-form').classList.add('show');
            } else {
                alert('Geçersiz kullanıcı adı veya şifre.');
            }
        }

        function adminLogin() {
            const username = document.getElementById('admin-username').value;
            const password = document.getElementById('admin-password').value;

            if (username === 'Halil İbrahim Deniz' && password === 'kemalataturk') {
                alert('Yönetici olarak giriş yapıldı.');
                // Yönetici işlevleri burada uygulanacak
            } else {
                alert('Geçersiz yönetici bilgileri.');
            }
        }

        function postComment() {
            const commentText = document.getElementById('comment-text').value;

            if (commentText.trim() !== '') {
                const commentSection = document.getElementById('comment-section');
                const newComment = document.createElement('div');
                newComment.classList.add('comment');

                const commentUser = document.createElement('p');
                commentUser.textContent = 'Kullanıcı Adı';

                const commentContent = document.createElement('p');
                commentContent.textContent = commentText;

                newComment.appendChild(commentUser);
                newComment.appendChild(commentContent);

                commentSection.appendChild(newComment);

                document.getElementById('comment-text').value = '';
            } else {
                alert('Yorum metni boş olamaz.');
            }
        }
    </script>
